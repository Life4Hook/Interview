//Создание таблицы в БД
CREATE TABLE Sportsman
(ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
 NAME VARCHAR(70),
 EMAIL VARCHAR(30),
 PHONE_NUMBER VARCHAR(11),
 BIRTHDAY DATE,
 AGE INT(3),
 CREATE_DATE DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
 PASSPORT_NUMBER INT(6),
 AVERAGE_RESULT FLOAT(5),
 BIOGRAPHY TEXT,
 VIDEO LONGBLOB)

 //Выбор топ 5 спортсменов
 //Если бы БД была заполнена по диаграмме то:
SELECT Sportsman.NAME, COUNT(Results.SPORTSMAN_ID) as `counter` FROM Results JOIN Sportsman ON Sportsman.ID = Results.SPORTSMAN_ID 
GROUP BY Results.SPORTSMAN_ID ORDER BY `counter` DESC LIMIT 5;
